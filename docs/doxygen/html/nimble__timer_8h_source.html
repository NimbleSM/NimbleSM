73  void Stop() {
74  using std::chrono::duration;
75  using std::chrono::duration_cast;
76  end_time_ = clock.now();
77  std::chrono::duration<double> time_increment(0.0);
78  if (end_time_ > start_time_) {
79  time_increment = duration_cast<duration<double>>(end_time_ - start_time_);
80  }
81  total_time_ += time_increment.count();
82  }
83

85  return total_time_;
86  }

234 std::string
235 GetModelMaterialParameters(int block_id) const
236 {
237 if (model_blocks_.find(block_id) == model_blocks_.end()) {
238 std::string none_str("none");
239 return none_str;
240 }
241 BlockProperties const& block = model_blocks_.at(block_id);
242 std::string const& material_key = block.material_key_;
243 std::string const& material_props = material_strings_.at(material_key);
244 return material_props;
245 }

247 int

249 {
250 int block_id = -1;
251 // std::map<int, BlockProperties>::iterator it;
252 // for(it = model_blocks_.begin(); it != model_blocks_.end();
253 // it++){
254 for (auto const& entry : model_blocks_) {
255 BlockProperties const& block_props = entry.second; // it->second;
256 if (material_key == block_props.material_key_) {
257 block_id = entry.first; // it->first; //Found the block id with the
258 // material string
259 break;

261 }

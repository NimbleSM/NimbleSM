812 elem_conn,

822 // non-zero on the diagonal

835 }
836 status = 1;
837 return status;
838 }

841 // Apply a line search
842 //
843
844 // evaluate residual for alpha = 1.0

856#endif

858 }
859 }
860 }
861 double trial_residual = ComputeQuasistaticResidual(
862 mesh,
863 data_manager,
864 bc,
865 linear_system_num_unknowns,
866 linear_system_global_node_ids,
867 time_previous,
868 time_current,
869 trial_displacement,
870 trial_internal_force,
871 trial_residual_vector.data(),
872 is_output_step);
873
874 //
875 // secant line search
876 //
877 double sr = nimble::InnerProduct(linear_solver_solution, residual_vector);
878 double s_trial_r = nimble::InnerProduct(linear_solver_solution, trial_residual_vector);
879 double alpha = -1.0 * sr / (s_trial_r - sr);
880
881 // evaluate residual for alpha computed with secant line search
882 for (int n = 0; n < linear_system_num_nodes; n++) {
883 std::vector<int> const& node_ids = map_from_linear_system[n];
884 for (auto const& node_id : node_ids) {
885 for (int dof = 0; dof < dim; dof++) {
886 int ls_index = n * dim + dof;
887#ifdef NIMBLE_DEBUG
888 if (ls_index < 0 || ls_index > linear_solver_solution.size() - 1) {
889 throw std::invalid_argument(
890 "\nError: Invalid index into linear solver solution vector in "
891 "QuasistaticTimeIntegrator().\n");
892 }
893#endif
894 displacement(node_id, dof) -= alpha * linear_solver_solution[ls_index];
895 }
896 }
897 }
898 residual = ComputeQuasistaticResidual(
899 mesh,
900 data_manager,
901 bc,
902 linear_system_num_unknowns,
903 linear_system_global_node_ids,
904 time_previous,
905 time_current,
906 displacement,
907 internal_force,
908 residual_vector.data(),
909 is_output_step);
910
911 // if the alpha = 1.0 result was better, use alpha = 1.0
912 if (trial_residual < residual) {
913 residual = trial_residual;
914 displacement.copy(trial_displacement);
915 internal_force.copy(trial_internal_force);
916 for (int i = 0; i < linear_system_num_unknowns; i++) { residual_vector[i] = trial_residual_vector[i]; }
917 }
918
919 iteration += 1;
920
921 if (my_rank == 0) {
922 std::cout << " iteration " << iteration << ": residual = " << residual
923 << ", linear cg iterations = " << num_cg_iterations << std::endl;
924 }
925
926 } // while (residual > convergence_tolerance ... )
927
928 if (iteration == max_nonlinear_iterations) {
929 if (my_rank == 0) {
930 std::cout << "\n**** Nonlinear solver failed to converge!\n" << std::endl;
931 std::cout << "**** Relevant input deck parameters for the nonlinear solver:" << std::endl;
932 std::cout << "**** nonlinear solver relative tolerance: " << parser.NonlinearSolverRelativeTolerance()
933 << std::endl;
934 std::cout << "**** nonlinear solver maximum iterations: " << parser.NonlinearSolverMaxIterations()
935 << std::endl;
936 }
937 status = 1;
938 return status;
939 } else {
940 if (is_output_step) data_manager.WriteOutput(time_current);
941 }
942
943 // swap states
944 model_data.UpdateStates(data_manager);
945 }
946
947 if (my_rank == 0) { std::cout << "\nComplete.\n" << std::endl; }
948
949 return status;
